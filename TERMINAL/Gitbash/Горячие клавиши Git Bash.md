# Горячие клавиши Git Bash

```bash
Tab                        # автозаполнение
Ctrl + a                   # переход в начало строки
Ctrl + e                   # возврат в конец строки
Ctrl + l                   # очищает экран от текста (clear)
↑                          # обращение к последней введённой команде
↓                          # обращение к предпоследней введённой команде

# При просмотре файла
f                          # вперед
b                          # назад
q                          # выйти
/...                       # поиск (нажать enter)
n                          # вперед по поиску
Shift + n                  # назад по поиску
```

# Горячие клавиши Vim

```bash
Shift + zz (:wq)           # Выйти из файла сохранив изменения (в командном режиме)
Shift + zq (:q!)           # Выйти из файла без сохранения изменений (в командном режиме)
```
