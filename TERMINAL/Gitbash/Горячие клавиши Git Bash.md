# Горячие клавиши Git Bash

```bash
Tab                        # автозаполнение
Ctrl + a                   # переход в начало строки
Ctrl + e                   # возврат в конец строки
Ctrl + l                   # очищает экран от текста (clear)
↑                          # обращение к последней введённой команде
↓                          # обращение к предпоследней введённой команде
&&                         # несколько команд

# При просмотре файла
f                          # вперед
b                          # назад
q                          # выйти
/...                       # поиск (подтвердить - enter)
n                          # вперед по поиску
Shift + n                  # назад по поиску
```

# Горячие клавиши Vim

```bash
# Режим команд
i                          # войти в режим ввода
Shift + zz (:wq)           # сохранить и выйти из файла
Shift + zq (:q!)           # выйти без сохранения
/...                       # поиск
n                          # вперед по поиску
Shift + n                  # назад по поиску

# Режим ввода (вставка)
Esc                        # вернуться в режим команд
```

# Горячие клавиши Nano

```bash
Ctrl + o                   # сохранить изменения (подтвердить - enter)
Ctrl + x                   # выйти
```
